180  void RigidBodyComponent::onUpdate(float timestep)
181  {
182  // If the body is dynamic
183  if (m_type == rp3d::BodyType::DYNAMIC || m_type == rp3d::BodyType::KINEMATIC)
184  {
185  // Find the interpolation factor using the accumulated time
186  rp3d::decimal interpolationFactor = *m_fpTimeAccumulator / m_fTimestep;
187  // Get the transform of the rigidbody and interpolate to find the transform to render
188  rp3d::Transform currentTransform = m_pBody->getTransform();
189  rp3d::Transform interpolatedTransform = rp3d::Transform::interpolateTransforms(
190  m_prevTransform, currentTransform, interpolationFactor);
191  // Get the position and rotation from the interpolated transfrom
192  rp3d::Vector3 pos = interpolatedTransform.getPosition();
193  rp3d::Quaternion rot = interpolatedTransform.getOrientation();
194 
195  // Get the euler angles of the rotation
196  glm::vec3 euler = glm::eulerAngles(glm::quat(rot.w, rot.x, rot.y, rot.z));
197 
198  // Send the interpolated transform to the position component
199  glm::vec3 posData = glm::vec3(pos.x, pos.y, pos.z);
200  ComponentMessage msg(ComponentMessageType::PositionSet, (void*)&posData);
201  sendMessage(msg);
202  glm::vec3 rotData = glm::vec3(glm::degrees(euler.x), glm::degrees(euler.y), glm::degrees(euler.z));
203  ComponentMessage msg2(ComponentMessageType::RotationSet, (void*)&rotData);
204  sendMessage(msg2);
205  // Save the current body transform
206  m_prevTransform = currentTransform;
207  }
208  }
