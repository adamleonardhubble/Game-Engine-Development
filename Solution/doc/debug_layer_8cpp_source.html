245  void DebugLayer::FPSprofiling(float timestep) // Calculate FPS profiling variables
246  {
247  m_fTimer += timestep; // Add the timestep to the time counter
248 
249  m_iTotalFramesPassed++; // Increment the total number of frames passed since application entry
250 
251  #pragma region Sample Size Management
252  if (m_vAverageFPS.size() > m_iSampleSize) // If the the size of the vector container is larger than 'm_iSampleSize', do the following
253  {
254  for (int i = 0; i <= m_vAverageFPS.size(); i++) // For the size of the vector container, do the following
255  {
256  m_vAverageFPS.pop_back(); // Pop the last element in the vector container, until the vector container is no loner larger than the sample size
257  }
258  }
259  #pragma endregion
260 
261  #pragma region Current and Average FPS
262  m_sTimestep = timestep * 1000.0f; // Set 'm_sTimestep' to the timestep divided by '1000' (convert to milliseconds)
263 
264  float FPS = (1 / timestep); // Calculate the FPS count of the current frame
265 
266  m_sFPSCurrent = FPS; // Set 'm_sFPSCurrent' to the current FPS count
267 
268  std::vector<float>::iterator iteratorFPS = m_vAverageFPS.begin(); // Vector container iterator, set to the first element in the vector container
269 
270  m_vAverageFPS.insert(iteratorFPS, FPS); // Insert the current FPS into the first element in the vector container
271 
272  for (int i = 0; i < m_vAverageFPS.size(); i++) // For the size of the vector container, do the following
273  {
274  m_fAccumulatedFPS += m_vAverageFPS.at(i); // Add and equal the FPS count at every element in the vector container
275  }
276 
277  float averageFPS = m_fAccumulatedFPS / m_vAverageFPS.size(); // Set 'averageFPS' to the average FPS count divided by the size of the vector
278 
279  m_sFPSAverage = averageFPS; // Set 'm_sFPSAverage' to the average FPS count
280 
281  m_fAccumulatedFPS = 0.0f; // Reset the average FPS
282  #pragma endregion
283 
284  #pragma region Dropped Frames Count
285  if (m_vAverageFPS.at(0) <= 0.1f * m_sFPSAverage) // If the current FPS count is '90%' smaller than the current average FPS count, do the following
286  {
287  m_iDroppedFramesCount++; // Increment the number of frames dropped
288  }
289 
290  m_fFrameDropAmount = (float)m_iDroppedFramesCount / (float)m_iTotalFramesPassed * 100; // Set 'm_fFrameDropAmount' to the percentage of frames dropped against the number of frames passed since runtime
291 
292  m_sDroppedFramesText = "Dropped Frames [" + std::to_string(m_iDroppedFramesCount) + "] \t Drop Percentage [" + std::to_string(m_fFrameDropAmount) + "] %"; // Set 'm_sDroppedFramesText' to the text format of the frame drop count and amount
293 
294  #pragma endregion
295 
296  #pragma region Update FPS Profiling Gradually
297  if (m_fTimer >= 0.1f) // If the time counter is larger than or equal to '0.1' (seconds), do the following
298  {
299  m_sCurrentFPSText = "Current FPS [" + std::to_string(m_sFPSCurrent) + "] \t Time Passed [" + std::to_string(m_sTimestep) + "] ms"; // Set 'm_sCurrentFPSText' to the text format of the current FPS count and timestep
300  m_sAverageFPSText = "Average FPS [" + std::to_string(m_sFPSAverage) + "]"; // Set 'm_sAverageFPSText' to the text format of the current average FPS count
301 
302  #pragma region FPS Line Graphs
303  m_fCurrentFPSCount = m_sFPSCurrent; // Set 'm_fCurrentFPSCount' to the most recent FPS count calculated
304  m_fCurrentAverageFPSCount = m_sFPSAverage; // Set 'm_fCurrentAverageFPSCount' to the most recent average FPS count calculated
305 
306  m_currentFrameGraphValues[m_iGraphIncrementer] = m_fCurrentFPSCount; // Set each element in the array to the most recent frame count, for every iteration
307  m_averageFrameGraphValues[m_iGraphIncrementer] = m_fCurrentAverageFPSCount; // Set each element in the array to the most recent average frame count, for every iteration
308 
309  if (m_iGraphIncrementer < 99) // If the number of elements in the arrays is smaller than the arrays size, do the following
310  {

312  }

319  }
