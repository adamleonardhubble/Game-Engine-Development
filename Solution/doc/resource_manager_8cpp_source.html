340  void ResourceManager::populateCharacters(std::unordered_multimap<std::string, unsigned int> fontsAndSizes)
341  {
342  FT_Library ft;
343  FT_Face face;
344  // Initialize freetype
345  if (FT_Init_FreeType(&ft))
346  LOG_CRITICAL("Could not start FreeType");
347 
348  unsigned char* texMemory; // Texture memory for characters
349  int memH = 1024; // Memory size
350  int memW = 1024;
351  int usedX = 0;
352  int usedY = 0;
353  // Allocate and clear memory for the texture
354  texMemory = (unsigned char*)malloc(memW * memH);
355  memset(texMemory, 0, memW * memH);
356 
357  std::map<std::pair<std::string, unsigned int>, std::pair<unsigned int, unsigned int>> maxSize; // Stores the maximum size for characters for each font and size
358  // For each size and font
359  for (auto& element : fontsAndSizes)
360  {
361  // Load the font
362  if (FT_New_Face(ft, element.first.c_str(), 0, &face))
363  LOG_CRITICAL("FreeType coudn't loat font: {0}", element.first);
364  // Set the font size
365  if (FT_Set_Pixel_Sizes(face, 0, element.second))
366  LOG_CRITICAL("FreeType couldn't set font face size of {0}", element.second);
367  // Add a value to the map to get the max sizes
368  maxSize.insert(std::make_pair(element, std::make_pair(0, 0)));
369  // For each ASCII value being used for characters
370  for (int i = s_ASCIIStart; i <= s_ASCIIEnd; i++)
371  {
372  // Load the character
373  if (FT_Load_Char(face, i, FT_LOAD_RENDER))
374  LOG_CRITICAL("Could not load the character {0}", i);
375  // If either of the sizes are bigger than the corresponding stored ones, change the stored one
376  maxSize[element].first = std::max(maxSize[element].first, face->glyph->bitmap.width);
377  maxSize[element].second = std::max(maxSize[element].second, face->glyph->bitmap.rows);
378  }
379  FT_Done_Face(face); // Done with the face
380  }
381 
382  unsigned int biggestInRow = 0;
383  // For each size and font
384  for (auto& element : fontsAndSizes)
385  {
386  // Load the font
387  if (FT_New_Face(ft, element.first.c_str(), 0, &face))
388  LOG_CRITICAL("FreeType coudn't loat font: {0}", element.first);
389  // Set the font size
390  if (FT_Set_Pixel_Sizes(face, 0, element.second))
391  LOG_CRITICAL("FreeType couldn't set font face size of {0}", element.second);
392  // Add the font and size to the map
393  s_characters.insert(std::make_pair(std::make_pair(parseFilePath(element.first), element.second), std::vector<Character>()));
394 
395  // For each ASCII value being used for characters
396  for (int i = s_ASCIIStart; i <= s_ASCIIEnd; i++)
397  {
398  // Load the character
399  if (FT_Load_Char(face, i, FT_LOAD_RENDER))
400  LOG_CRITICAL("Could not load the character {0}", i);
401  // Set the character values and add it to the vector in the map
402  s_characters[std::make_pair(parseFilePath(element.first), element.second)].push_back(Character(
403  glm::ivec2(face->glyph->bitmap.width, face->glyph->bitmap.rows),
404  glm::ivec2(face->glyph->bitmap_left, face->glyph->bitmap_top),
405  face->glyph->advance.x,
406  glm::vec2((float)usedX / (float)memW, (float)usedY / (float)memH),
407  glm::vec2(((float)usedX + (float)face->glyph->bitmap.width) / (float)memW, ((float)usedY + (float)face->glyph->bitmap.rows) / (float)memH)
408  ));
409  // Get the face buffer (texture data)
410  unsigned char* buffer = face->glyph->bitmap.buffer;
411  // For each row
412  for (int i = 0; i < face->glyph->bitmap.rows; i++)
413  {
414  if (usedY + maxSize[element].second > memH)
415  {
416  LOG_CRITICAL("Can't fit all characters on the texture, use smaller character sizes and scale");
417  return;
418  }
419  // Copy the row from the buffer to the new texture
420  memcpy(texMemory + (((memW * usedY) + (i * memW)) + usedX), buffer + (i * face->glyph->bitmap.width), face->glyph->bitmap.width);
421  }
422  biggestInRow = glm::max(biggestInRow, maxSize[element].second); // Always be equal to the tallest character that has been added to a row
423 
424  // Increase the used x value
425  usedX += face->glyph->bitmap.width;
426  // If reached the edge
427  if (usedX + maxSize[element].first > memW)
428  {
429  usedX = 0; // Reset used x
430  usedY += biggestInRow; // Move down by the size of the tallest character in the row
431  biggestInRow = 0;
432  }
433  }
434  FT_Done_Face(face); // Done with the face
435  }
436 
437  FT_Done_FreeType(ft); // Done with freetype
438  // Create a texture from the data created
439  s_fontTexture.reset(Texture::createFromRawData(memW, memH, 1, texMemory));
440  }
