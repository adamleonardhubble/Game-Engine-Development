98  void Free3DCameraController::updateView()
99  {
100  // Calculate a new forward vector from the yaw and the pitch
101  glm::vec3 forward;
102  forward.x = sin(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
103  forward.y = sin(glm::radians(m_fPitch));
104  forward.z = cos(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
105 
106  m_forward = glm::normalize(forward);
107 
108  // Calculate right and up vectors
109  m_right = glm::normalize(glm::cross(m_forward, m_worldUp));
110  m_up = glm::normalize(glm::cross(m_right, m_forward));
111 
112  // Calculate the view
113  m_view = glm::lookAt(m_position, m_position + m_forward, m_up);
114  // Set the view in the camera
115  m_camera->setViewAndPos(m_view, m_position);
116  }
