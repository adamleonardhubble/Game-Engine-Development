109  void FreeMoving3DCameraComponent::updateView()
110  {
111  // Calculate a new forward vector from the yaw and the pitch
112  glm::vec3 forward;
113  forward.x = sin(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
114  forward.y = sin(glm::radians(m_fPitch));
115  forward.z = cos(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
116 
117  m_forward = glm::normalize(forward);
118 
119  // Calculate right and up vectors
120  m_right = glm::normalize(glm::cross(m_forward, m_worldUp));
121  m_up = glm::normalize(glm::cross(m_right, m_forward));
122 
123  // Calculate the view
124  m_view = glm::lookAt(m_position, m_position + m_forward, m_up);
125  // Set the view in the camera
126  m_camera->setViewAndPos(m_view, m_position);
127  }
