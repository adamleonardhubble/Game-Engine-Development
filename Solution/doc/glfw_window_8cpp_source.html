248  {
249  m_properties.m_isFullScreen = fullscreen; // Set the m_isFullScreen property
250  // Get the current video mode of the monitor
251  const GLFWvidmode* mode = glfwGetVideoMode(m_pMonitor);
252  // If fullscreen is being turned on
253  if (fullscreen)
254  {
255  // Set the current window size and position to the stored positions

259  // Makes the window fullscreen
260  glfwSetWindowMonitor(m_pNativeWindow, m_pMonitor, 0, 0, mode->width, mode->height, mode->refreshRate);
261  }
262  else // If fullscreen is being turned off
263  {
264  // Set the window back to the stored size and position
265  glfwSetWindowMonitor(m_pNativeWindow, nullptr, m_properties.m_storedXPos, m_properties.m_storedYPos, m_properties.m_storedWidth, m_properties.m_storedHeight, mode->refreshRate);
