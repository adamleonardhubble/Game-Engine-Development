101  void CameraController3rdPerson::updateView()
102  {
103  // Calculate a new forward vector from the yaw and the pitch
104  glm::vec3 forward;
105  forward.x = sin(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
106  forward.y = sin(glm::radians(m_fPitch));
107  forward.z = cos(glm::radians(m_fYaw)) * cos(glm::radians(m_fPitch));
108 
109  m_forward = glm::normalize(forward);
110 
111  // Calculate right and up vectors
112  m_right = glm::normalize(glm::cross(m_forward, m_worldUp));
113  m_up = glm::normalize(glm::cross(m_right, m_forward));
114 
115  // Calculate camera position based on player position and offset
116  m_position = m_playerPosition + glm::vec3((m_forward * m_playerOffset.z) + (m_up * m_playerOffset.y) + (m_right * m_playerOffset.x));
117  // Calculate the view
118  m_view = glm::lookAt(m_position, m_position + m_forward, m_up);
119  // Set the view in the camera
120  m_camera->setViewAndPos(m_view, m_position);
121  }
